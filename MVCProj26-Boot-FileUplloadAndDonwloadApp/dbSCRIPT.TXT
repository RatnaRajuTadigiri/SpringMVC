CREATE TABLE "SYSTEM"."EMPLOYEE_UPLOAD" 
   (	"EID" NUMBER(10,0) NOT NULL ENABLE, 
	"ENAME" VARCHAR2(20 BYTE), 
	"EADD" VARCHAR2(20 BYTE), 
	"RESUMELOCATION" VARCHAR2(20 BYTE),
