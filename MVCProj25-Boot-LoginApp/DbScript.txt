CREATE OR REPLACE PROCEDURE P_AUTHENTICATION 
(
  UNAME IN VARCHAR2 
, PWD IN VARCHAR2 
, RESULT OUT VARCHAR2 
) AS 
  CNT NUMBER(5);
BEGIN
  SELECT  COUNT(*)INTO CNT FROM USERSLIST  WHERE USERNAME=UNAME AND PASSWORD=PWD;
  IF(CNT<>0)THEN
   RESULT:='VALID CRENETILAS';
   ElSE
   RESULT:='INVALID CREDENTIALS';
   END IF;
END P_AUTHENTICATION;
