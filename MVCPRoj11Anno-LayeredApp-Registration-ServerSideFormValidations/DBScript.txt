CREATE SEQUENCE  "SYSTEM"."REGID_SEQ"  MINVALUE 1 MAXVALUE 10000000 INCREMENT BY 1 START WITH 1 CACHE 20 NOORDER  NOCYCLE ;

CREATE TABLE "SYSTEM"."CORONA_VACCINE" 
   (	"REGID" NUMBER(20,0) NOT NULL ENABLE, 
	"NAME" VARCHAR2(20 BYTE), 
	"ADDRS" VARCHAR2(20 BYTE), 
	"ADHARNO" NUMBER(14,0), 
	"PRICE" FLOAT(126), 
	"INCOME" FLOAT(126), 
	"VACCINE_NAME" VARCHAR2(20 BYTE), 
	"REMARKS"  VARCHAR2(50 BYTE), 
	 CONSTRAINT "CORONA_VACCINE_PK" PRIMARY KEY ("REGID"))
